2
Copia la URL del despliegue y pégala aquí:
3
Completa el sorteo de una categoría y pulsa ENVIAR FIXTURE A SHEETS — los partidos se escribirán automáticamente en tu hoja con el formato correcto.
4
En Sheets solo editas Goles Local (col F) y Goles Visitante (col G). Score y Resultado se calculan solos.
Pega este código en Apps Script de tu Google Sheets:
// ============================================ // COPA CAJAMARCA — Apps Script // ============================================ const SHEET_FIXTURE = "FIXTURE"; // Nombre de tu hoja function doGet(e) { const action = e.parameter.action || ""; if (action === "writeFixture") return writeFixture(e); return ContentService.createTextOutput("Copa Cajamarca API OK"); } function writeFixture(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_FIXTURE); if (!sheet) sheet = ss.insertSheet(SHEET_FIXTURE); const payload = JSON.parse(e.parameter.data); const categoria = payload.categoria; const partidos = payload.partidos; // Encabezados (solo si la hoja está vacía) if (sheet.getLastRow() === 0) { sheet.appendRow(["Jornada","CATEGORIA","Local","Score","Visitante","Goles Local","Goles Visitante","Resultado"]); sheet.getRange(1,1,1,8).setFontWeight("bold").setBackground("#1a3a6b").setFontColor("#ffffff"); sheet.setFrozenRows(1); } // Borrar filas previas de esta categoría const data = sheet.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][1]) === String(categoria)) sheet.deleteRow(i + 1); } // Insertar nuevos partidos partidos.forEach(p => { sheet.appendRow([ p.jornada, categoria, p.local, "- : -", // Score inicial p.visitante, "", // Goles Local (vacío para editar) "", // Goles Visitante (vacío para editar) "PENDIENTE" ]); }); // Formato de la última columna (Resultado) en negrita const last = sheet.getLastRow(); if (last > 1) { sheet.getRange(2, 8, last - 1, 1).setFontWeight("bold"); sheet.getRange(2, 6, last - 1, 2).setBackground("#e8f5e9"); // verde claro para goles sheet.autoResizeColumns(1, 8); } return ContentService .createTextOutput(JSON.stringify({ ok: true, rows: partidos.length })) .setMimeType(ContentService.MimeType.JSON); }
